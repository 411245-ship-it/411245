import random # 1. 建立單字庫 (可以自行增加) word_bank = { "apple": "蘋果", "banana": "香蕉", "challenge": "挑戰", "opportunity": "機會", "persistent": "堅持不懈的", "algorithm": "演算法", "efficiency": "效率" } def start_quiz(): words = list(word_bank.keys()) random.shuffle(words) # 打亂順序 score = 0 total = len(words) print("--- 英文單字測驗開始 ---") print(f"共有 {total} 題，輸入 'q' 可以隨時退出。\n") for i, word in enumerate(words, 1): answer = input(f"第 {i} 題：[{word}] 的中文是什麼？ ").strip() if answer.lower() == 'q': break if answer == word_bank[word]: print("✅ 答對了！") score += 1 else: print(f"❌ 答錯了，正確答案是：{word_bank[word]}") print("-" * 20) print(f"\n測驗結束！你答對了 {score}/{total} 題。") print(f"得分率：{(score/total)*100:.1f}%") if __name__ == "__main__": start_quiz()
